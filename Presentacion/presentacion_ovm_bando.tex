\begin{frame}{Definicion del sistema}
  \begin{itemize}
    \item Se modela una \textbf{via circular} de longitud $L$ con $N$ vehiculos.
    \item Cada vehiculo ajusta su velocidad segun la \textbf{separacion} respecto al vehiculo delantero.
    \item El fenomeno de interes es la \textbf{congestion fantasma}: una pequena perturbacion local puede propagarse como onda de frenado.
    \item El objetivo no es reproducir una avenida real, sino estudiar de forma controlada la \textbf{estabilidad del flujo vehicular}.
  \end{itemize}
\end{frame}

\begin{frame}{Modelo matematico}
  Para cada vehiculo $i$ se considera:
  \[
    x_i''(t) = a\left(V(h_i(t)) - x_i'(t)\right),
  \]
  donde:
  \begin{itemize}
    \item $x_i(t)$ es la posicion del vehiculo $i$.
    \item $x_i'(t)=v_i(t)$ es su velocidad.
    \item $a$ mide la rapidez de reaccion del conductor.
    \item $h_i(t)$ es la separacion con el vehiculo delantero.
  \end{itemize}

  La velocidad optima usada fue:
  \[
    V(h) = \tanh(h-2) + \tanh(2).
  \]
\end{frame}

\begin{frame}{Separacion y parametros}
  En la via circular se usa:
  \[
    h_i = x_{i+1} - x_i, \qquad
    h_N = x_1 + L - x_N.
  \]

  Configuracion base usada en las comparaciones:
  \begin{itemize}
    \item $N = 20$ vehiculos.
    \item $L = 40$.
    \item $a = 1.0$.
    \item $t_0 = 0$, $t_f = 40$.
    \item Paso $\Delta t \in \{0.20, 0.10, 0.05\}$.
    \item Perturbacion inicial base: $0.20$.
  \end{itemize}
\end{frame}

\begin{frame}{Reduccion a sistema de primer orden}
  Para resolver numericamente, definimos
  \[
    v_i = x_i'.
  \]

  Entonces el sistema queda:
  \[
    x_i' = v_i,
    \qquad
    v_i' = a\left(V(h_i) - v_i\right).
  \]

  El vector de estado es:
  \[
    y = [x_1, x_2, \dots, x_N, v_1, v_2, \dots, v_N].
  \]

  Para $N=20$, el problema equivale a \textbf{40 ecuaciones de primer orden}.
\end{frame}

\begin{frame}{Metodos numericos e implementacion}
  \begin{itemize}
    \item Los metodos implementados directamente por el equipo fueron:
    \begin{itemize}
      \item Euler explicito.
      \item Heun / RK2.
      \item Runge-Kutta clasico de cuarto orden (RK4).
    \end{itemize}
    \item RK4 se tomo como metodo principal por su mejor precision.
    \item RK45 de \texttt{SciPy} se uso unicamente como \textbf{referencia de validacion}.
    \item El proyecto se implemento en Python con \texttt{NumPy}, \texttt{SciPy}, \texttt{Matplotlib} y \texttt{Streamlit}.
  \end{itemize}
\end{frame}

\begin{frame}{Validacion y analisis del error}
  \begin{table}
    \centering
    \scriptsize
    \begin{tabular}{lllll}
      \toprule
      Metodo & $\Delta t$ & Error prom. & Error final & Tiempo (s) \\
      \midrule
      Euler & 0.20 & $2.07\times10^{-1}$ & $7.88\times10^{-1}$ & 0.0080 \\
      Heun  & 0.20 & $4.97\times10^{-3}$ & $1.75\times10^{-2}$ & 0.0124 \\
      RK4   & 0.20 & $6.35\times10^{-6}$ & $1.96\times10^{-5}$ & 0.0314 \\
      Euler & 0.05 & $3.10\times10^{-2}$ & $1.23\times10^{-1}$ & 0.0246 \\
      Heun  & 0.05 & $3.12\times10^{-4}$ & $1.11\times10^{-3}$ & 0.0534 \\
      RK4   & 0.05 & $2.76\times10^{-8}$ & $8.73\times10^{-8}$ & 0.0981 \\
      \bottomrule
    \end{tabular}
  \end{table}

  \vspace{0.3cm}
  \begin{itemize}
    \item Al disminuir $\Delta t$, el error baja en los tres metodos.
    \item RK4 presenta la mejor coincidencia con RK45.
  \end{itemize}
\end{frame}

\begin{frame}{Lectura de los escenarios}
  \begin{itemize}
    \item \textbf{Flujo uniforme}: la dispersion permanece cerca de cero; el sistema conserva el equilibrio.
    \item \textbf{Perturbacion fuerte y varias perturbaciones}: generan ondas de frenado mas visibles.
    \item \textbf{Reaccion lenta}: amplifica mas la dispersion; es el caso mas inestable.
    \item \textbf{Reaccion alta}: amortigua la perturbacion y acerca el sistema a un flujo mas uniforme.
    \item \textbf{Alta densidad}: menos espacio entre vehiculos favorece la propagacion de diferencias de velocidad.
  \end{itemize}
\end{frame}

\begin{frame}{Conclusiones}
  \begin{itemize}
    \item El parametro de reaccion $a$ es clave para la estabilidad del flujo.
    \item La magnitud y cantidad de perturbaciones iniciales modifican la intensidad de las oscilaciones.
    \item Menor separacion promedio entre vehiculos favorece la propagacion de ondas de congestion.
    \item RK4 fue el metodo implementado que mejor reprodujo la referencia RK45.
    \item El proyecto permitio conectar modelado, reduccion a primer orden, simulacion y analisis numerico en un solo caso de estudio.
  \end{itemize}
\end{frame}

\begin{frame}{Aprendizajes del proyecto}
  \begin{itemize}
    \item Un sistema con reglas locales simples puede producir comportamiento colectivo complejo.
    \item Las visualizaciones ayudan a validar si los resultados numericos tienen sentido fisico.
    \item El analisis del error no es un requisito aislado: cambia directamente la confianza en la interpretacion del modelo.
    \item Integrar modelado, codigo y presentacion exige coordinacion constante del equipo.
  \end{itemize}
\end{frame}
